#include<stdio.h>
#include<stdlib.h> 
#define OVERFLOW 0 
#define ERROR 0
#define OK  1
#define Length 10
typedef int Status;
typedef int ElemType;
 
typedef struct LNode{
    ElemType data;
    struct LNode *next;
}LNode,*LinkList;
////////////////////////////////打印单链表//////////////////////////////////////////////////////
void printAll(LinkList L){    //打印长度为n的单链表
    LinkList p=L;
    printf("\n");
    for(int i=0;p->next!=NULL;i++){
        printf("%d\t",p->next->data);
        p=p->next;
    }
   
    printf("\n");
}
 
////////////////////////////////建立单链表///////////////////////////////////////////////////
void CreateList_L(LinkList &L,int n){    //建立长度为n的单链表
    L=(LinkList)malloc(sizeof(LNode));
    L->next=NULL;    //先建立一个带头结点的单链表                  之所以赋个NULL是为了让遍历方便，否则无法判断 什么时候该停止遍历
   
    for(int i;i<n;i++){    //头插法建立单链表
        LinkList p=(LinkList)malloc(sizeof(LNode));
        p->next=L->next;
        L->next=p;
       
    }
    printf("单链表创建成功！") ;
 
   
}
 
 
////////////////////////////////得到指定元素/////////////////////////////////////////////////////////
ElemType GetElem_L(LinkList &L,int i){    //得到单链表的第i个元素
    ElemType e;
    LinkList p=L;
    for(int j=0;j!=i;j++){
        p=p->next;
    }
    e=p->data;
   
    return e;
}
 
////////////////////////////////插入元素//////////////////////////////////////////////////////////////
Status ListInsert_L(LinkList &L,int i,ElemType e){    //在单链表的第i个位置插入元素e
    LinkList p=L;
     LinkList q=(LinkList)malloc(sizeof(LNode));
    
    for(int j=0;j!=i-1;j++){
        p=p->next;
    }
    q->next=p->next;
    p->next=q;
    q->data=e;
   
    return OK;
}
 
 
////////////////////////////////删除元素//////////////////////////////////////////////////////////
Status ListDelete_L(LinkList &L,int i){    //删除单链表第i个元素
    LinkList p=L;
   
    for(int j=0;j!=i-1;j++){
        p=p->next;
    }
    LinkList q=p->next;
    p->next=p->next->next;
    free(q);
   
    return OK;
}
 
 
////////////////////////////////倒置单链表/////////////////////////////////////////////////////////
void ListInvert_L(LinkList &L){
    LinkList p=L,q=L;
    int length;
   
    for(length=0;p->next!=NULL;length++){        //先测一下单链表的长度
        p=p->next;
    }                   
                   
                                        
    for(int i=0;i<length/2;i++){    
        p=L;
        q=q->next;
        for(int j=0;j<length-i;j++){    //找到需要进行交换的结点
            p=p->next;
        }
       
        ElemType t=q->data;        //进行交换
        q->data=p->data;
        p->data=t;
                                       
    }
   
   
}
/////////更好的倒置方法////////////////
 
void ReverseList(LinkList &L){ 
   
    LinkList p, q, r;  //p是当前结点    //q是下一个结点        //r是上一个结点
     
    p = L->next; 
    q=r=NULL; 
 
    while(p) 
    { 
       
    q = p->next;      //备份下一个结点的地址
    p->next = r;      //让当前结点的指针指向上一个结点         第一次执行本语句是将第一个结点的next域设为NULL方便遍历
    r = p; 
    p = q; 
   
    } 
   
    L->next=r;        //头结点指向原单链表的最后一个结点
   
   
   
} 
 
////////////////////////////////合并单链表///////////////////////////////////////////////////////
LinkList MergeList_L(LinkList &La,LinkList &Lb) {
    //已知单链表La和Lb的元素按值递增排列
    //归并La和Lb得到新的单链表Lc，Lc的元素也按值递增排列
    LinkList Lc=(LinkList)malloc(sizeof(LNode));
    LinkList p=La->next;
    LinkList q=Lb->next;
    LinkList r=Lc;
    //Lc->next=NULL;    //先建立一个带头结点的单链表                  之所以赋个NULL是为了让遍历方便，否则无法判断 什么时候该停止遍历
    while(p&&q){
        if(p->data<q->data){
            r->next=p;
            p=p->next;
            r=r->next;
       
        }
        else{
            r->next=q;
            q=q->next;
            r=r->next;
        }   
    }
   
    r->next=p?p:q;    //因为上面的while语句内部 总会使 要合并的两个链表中的一个剩下一个结点（因为此时连一个链表已经遍历到了最后一个元素，其next域为NULL，使算法停止遍历） 所以需要这个语句来将最后一个结点链接上来
   
   
   
    return Lc;
}

    printAll(L);
   
    ////////////////////////////////倒置单链表////////////////////////////////////
    printf("\n----------------------------------倒置单链表---------------------------------------\n");
    //ListInvert_L(L);

    printf("\n----------------------------------删除单链表第s个元素---------------------------------------\n");
    int s=5;

    ListDelete_L(L,s);   
    printAll(L);
 
   
    printf("\n----------------------------------合并单链表---------------------------------------\n");
    LinkList La,Lb;
    CreateList_L(La,Length);
   
    LinkList p1=La;
    for(int i=0;i<Length;i++){    //自动为单链表添数
        p1->next->data=i*2;
        p1=p1->next;
    }
    printf("\n单链表La为：\n");
    printAll(La);
   
    CreateList_L(Lb,Length);
   
    LinkList p2=Lb;
    for(int i=0;i<Length;i++){    //自动为单链表添数
        p2->next->data=i*2+1;
        p2=p2->next;
    }
    printf("\n单链表Lb为：\n");
    printAll(Lb);
   
    LinkList Lm=MergeList_L(La,Lb);
    printAll(Lm);
   
   
   
   
   
   
   
    system("pause");

}
