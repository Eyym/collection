#include<stdio.h>
#include<stdlib.h>
 
#define STACK_INIT_SIZE 100        //存储空间初始分配量 
#define STACKINCREMENT 10        //存储空间分配增量 
#define OVERFLOW 0  
#define ERROR 0
#define OK 1
 
typedef int Status;
typedef int SElemType;
typedef struct{
    SElemType *base;        //在栈构造之前和销毁之后，base为NULL 
    SElemType *top;            //栈顶指针 
    int stacksize;            //当前已分配的存储空间，以元素为单位 
}SqStack;
 
 
//------------------------栈顶初始化------------------------------------
Status InitStack(SqStack &S){        //构造一个空栈 
    S.base=(SElemType *)malloc(STACK_INIT_SIZE*sizeof(SElemType));
    if(!S.base){
        exit(OVERFLOW);        //存储分配失败
    }
    S.top =S.base;
    S.stacksize=STACK_INIT_SIZE;
    return OK;
} 
 
//------------------------得到栈顶元素------------------------------------
Status GetTop(SqStack S,SElemType &e){        //若栈不空，则用e返回S的栈顶元素，并返回OK，否则返回ERROR 
    if(S.top==S.base){
        return ERROR;
    }
    e=*(S.top-1);
    return OK;
} 
 
//------------------------压栈------------------------------------
Status Push(SqStack &S,SElemType e){        //插入元素e为新的栈顶元素 
    if(S.top-S.base>=S.stacksize){        //栈满，追加存储空间 
        S.base=(SElemType *)realloc(S.base,(S.stacksize+STACKINCREMENT)*sizeof(SElemType));
        if(!S.base){        //存储分配失败 
            exit(OVERFLOW);
        }
        S.top=S.base+S.stacksize;       //最后一个元素的下一个元素的位置
        S.stacksize+=STACKINCREMENT;
    }
    *S.top++=e;
    return OK;
}
 
 
 
//------------------------弹栈------------------------------------
Status Pop(SqStack &S,SElemType &e){        //若栈不空，则删除S的栈顶元素，用e返回其值，并返回OK，否则返回ERROR
    if(S.top==S.base){
        return ERROR;
    } 
    e=*--S.top;
    return OK;
    
}
 
 
//------------------------毁栈------------------------------------
Status DestroyStack(SqStack &S){
    free(S.base);
    free(S.top);
    S.base=NULL;
    printf("\n栈已被成功摧毁！\n\n");
    return OK;
}
 
//------------------------清空栈------------------------------------
Status ClearStack(SqStack &S){
    S.top=S.base;
    return OK;
}
//------------------------确认栈是否为空-----------------------------------
Status StackEmpty(SqStack S){
    if(S.base==S.top){
    return 1;
    }
    return 0;
} 
 
 
//------------------------计算栈的长度------------------------------------
int StackLength(SqStack S){
    if(S.base==S.top){
    return 0;
    }
    return S.top-S.base;
}
 
//------------------------遍历栈------------------------------------
Status StackTraverse(SqStack S){//从栈底到栈顶的方向
    printf("\n此时栈内情况为：\n");
    if(S.top==S.base){
    return 0;
    }
    while(S.base <S.top ){
    printf("%d\t",*(S.base++));
    }
    printf("\n");
    return 1;
}
 
//----------------------------------主函数---------------------------------------
int main(){
    SqStack S;
    printf("\n---------------------初始化栈并通过“压栈”添加十个元素--------------------------\n");
    InitStack(S);   //初始化栈
    
    for(int i=0;i<10;i++){
        Push(S,i);      //添加10个元素
    }
    
    StackTraverse(S);   //遍历一下
//---------------------------------------------
    printf("\n---------------------通过“弹栈”删除4个元素--------------------------\n");
    SElemType e;
    for(int i=0;i<4;i++){
        Pop(S,e);      //删除4个元素
        printf("\n已删除%d\n",e);
    }
    StackTraverse(S);   //遍历一下
    
//---------------------------------------------
    printf("\n---------------------栈顶元素为--------------------------\n");
    GetTop(S,e);
    printf("此时栈顶元素  %d\n",e);
    
    
//---------------------------------------------
    printf("\n---------------------判断栈是否为空--------------------------\n");
    int p=StackEmpty(S);
    if(p==0){
        printf("\n栈不为空\n");
    }
    else{
        printf("\n栈为空\n");
    }
    
//---------------------------------------------
    printf("\n---------------------获取栈的长度--------------------------\n");
    
    printf("\n栈的长度为  %d\n",StackLength(S));
    
//---------------------------------------------
    printf("\n---------------------清空栈--------------------------\n");
    ClearStack(S);
    StackTraverse(S);   //遍历一下
//---------------------------------------------
    printf("\n---------------------摧毁栈--------------------------\n");
    DestroyStack(S);
 
    
    
    
    
    
    
    
    system("pause");
    return 0;
}
