import java.sql.*;
 
 
class Test {
    public static void main(String[] args) throws SQLException {
        Test.readStatement("虎子");
        Test.readPreparedStatement("张三");
    }
 
    //使用PreparedStatement获取数据
    static void readPreparedStatement(String name) throws SQLException {
        Connection conn = null;
        PreparedStatement ps = null;
        ResultSet rs = null;
        try {
            // 2.建立连接
            conn = JdbcUtils.getConnection();
 
            // 3.创建语句
            String sql = "select id, name, money, birthday from user where name=?";
            ps = conn.prepareStatement(sql);
            ps.setString(1, name);  //替换sql语句中的第1个“?”为形参name的值
            // 4.执行语句
            rs = ps.executeQuery();
 
            // 5.处理结果
            while (rs.next()) {
                System.out.println(rs.getInt("id") + "\t"
                        + rs.getString("name") + "\t"
                        + rs.getDate("birthday") + "\t"
                        + rs.getFloat("money"));
            }
 
        } finally {
            JdbcUtils.free(rs, ps, conn);
        }
    }
 
    //使用Statement获取数据
    static void readStatement(String name) throws SQLException {
        Connection conn = null;
        Statement st = null;
        ResultSet rs = null;
        try {
            // 2.建立连接
            conn = JdbcUtils.getConnection();
            // conn = JdbcUtilsSing.getInstance().getConnection();
 
            // 3.创建语句
            String sql = "select id, name, money, birthday  from user where name='"
                    + name + "'";
            st = conn.createStatement();
            // 4.执行语句
            rs = st.executeQuery(sql);
 
            // 5.处理结果
            while (rs.next()) {
                System.out.println(rs.getObject("id") + "\t"
                        + rs.getObject("name") + "\t"
                        + rs.getObject("birthday") + "\t"
                        + rs.getObject("money"));
            }
        } finally {
            JdbcUtils.free(rs, st, conn);
        }
    }
 
}
