// ==UserScript==
// @name         btlibrary屏蔽广告
// @namespace    http://tampermonkey.net/
// @version      1.0
// @description  屏蔽btlibrary的广告
// @author       loubth
// @match        *btlibrary.bid/*
// @require      http://ajax.aspnetcdn.com/ajax/jQuery/jquery-1.7.2.min.js
// @grant        none

// ==/UserScript==

//去除垃圾链接
setInterval('$(".wrap-content").unbind("click");', 1000);
//去广告
$("<style id='myStyle'></style>").prependTo($("head"));
$("#myStyle").html("[id^=__jclm],[id^=__dsje],.shares,iframe{display:none!important;}");
